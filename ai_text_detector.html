# ── Rabin-Karp Rolling Hash Implementation ──────────────────────────────────
# Uses polynomial rolling hash to efficiently hash text n-grams.
# Each new hash is computed in O(1) by "rolling" off the old character.

class RollingHash:
    def __init__(self, base=31, mod=100_003):
        self.base = base          # polynomial base (usually prime)
        self.mod  = mod           # large prime modulus to avoid overflow

    def hash_ngram(self, tokens: list[str]) -> int:
        """Compute polynomial hash for a list of word tokens."""
        h = 0
        for i, token in enumerate(tokens):
            token_val = self._str_hash(token)
            h = (h + token_val * pow(self.base, i, self.mod)) % self.mod
        return h

    def _str_hash(self, s: str) -> int:
        """Simple polynomial hash for a single word."""
        h = 0
        for ch in s:
            h = (h * 31 + ord(ch)) % self.mod
        return h + 1   # +1 avoids zero multiplication

    def rolling_hash_all(self, tokens: list[str], n: int) -> list[int]:
        """
        Generate hash values for all n-grams using the sliding window.
        Rolling update: remove contribution of outgoing token, shift,
        add contribution of incoming token — O(1) per step.
        """
        if len(tokens) < n:
            return []

        token_hashes = [self._str_hash(t) for t in tokens]
        base_pow_n   = pow(self.base, n - 1, self.mod)

        # Compute first window
        h = 0
        for i in range(n):
            h = (h + token_hashes[i] * pow(self.base, i, self.mod)) % self.mod
        hashes = [h]

        # Roll the window
        for i in range(1, len(tokens) - n + 1):
            h = (h - token_hashes[i - 1]) % self.mod  # remove old head
            h = (h * pow(self.base, -1, self.mod)) % self.mod # shift right
            h = (h + token_hashes[i + n - 1] * base_pow_n) % self.mod # add new tail
            hashes.append(h)

        return hashes

# ── AI Text Detector ─────────────────────────────────────────────────────────
import re
from collections import Counter
from rabin_karp import RollingHash

def preprocess(text: str) -> list[str]:
    """Lowercase, remove punctuation, split into word tokens."""
    text = text.lower()
    text = re.sub(r"[^a-z0-9\s]", "", text)
    return text.split()

def compute_fingerprint(text: str, n=3, base=31, mod=100_003) -> dict:
    """
    Build a hash fingerprint for a text sample.
    Returns:
      - hashes      : all rolling hash values
      - counts      : Counter of hash → frequency
      - unique_ratio: unique hashes / total hashes
      - repeat_rate : fraction of hashes that appear >1 time
      - top_ngrams  : most repeated n-gram strings
    """
    rh     = RollingHash(base, mod)
    tokens = preprocess(text)
    hashes = rh.rolling_hash_all(tokens, n)

    if not hashes:
        return {}

    counts       = Counter(hashes)
    unique_ratio = len(counts) / len(hashes)
    repeated     = sum(1 for c in counts.values() if c > 1)
    repeat_rate  = repeated / len(counts)

    # Map hashes back to their n-gram strings
    ngram_texts = [" ".join(tokens[i:i+n]) for i in range(len(tokens)-n+1)]
    ngram_count = Counter(ngram_texts)
    top_ngrams  = ngram_count.most_common(10)

    return dict(
        hashes=hashes, counts=counts,
        unique_ratio=unique_ratio, repeat_rate=repeat_rate,
        top_ngrams=top_ngrams, total=len(hashes),
        unique=len(counts),
    )

def score_ai_likelihood(target_fp, human_fp, ai_fp) -> float:
    """
    Score 0–1: 0 = very human-like, 1 = very AI-like.
    Strategy: AI text has higher repeat_rate and lower unique_ratio
    compared to human text. We interpolate the target's position.
    """
    # Feature deltas
    human_rr = human_fp["repeat_rate"]
    ai_rr    = ai_fp["repeat_rate"]
    tgt_rr   = target_fp["repeat_rate"]

    human_ur = human_fp["unique_ratio"]
    ai_ur    = ai_fp["unique_ratio"]
    tgt_ur   = target_fp["unique_ratio"]

    # Normalize target position between human and AI poles
    rr_score = _normalize(tgt_rr, human_rr, ai_rr)   # higher = more AI
    ur_score = _normalize(ai_ur,   tgt_ur,  human_ur)  # lower unique = more AI

    return (rr_score * 0.5 + ur_score * 0.5)

def _normalize(val, lo, hi) -> float:
    if abs(hi - lo) < 1e-9: return 0.5
    return max(0, min(1, (val - lo) / (hi - lo)))

# ── Usage Example ─────────────────────────────────────────────────────────────
from detector import compute_fingerprint, score_ai_likelihood

human_text = """
I honestly didn't know what to expect when I first started college.
Everyone kept telling me it'd be the best years of my life, but at first
it just felt overwhelming and kind of lonely, if I'm being honest.
"""

ai_text = """
The transition to college life presents numerous opportunities for personal
and academic development. Students are encouraged to engage actively with
campus resources and to build meaningful connections with their peers.
"""

unknown_text = """
Starting college can be both exciting and challenging. It is important for
students to make use of available support systems and to maintain a healthy
work-life balance throughout the academic year.
"""

# Compute fingerprints
N = 3   # trigrams
human_fp  = compute_fingerprint(human_text, n=N)
ai_fp     = compute_fingerprint(ai_text, n=N)
target_fp = compute_fingerprint(unknown_text, n=N)

# Score
score = score_ai_likelihood(target_fp, human_fp, ai_fp)

print(f"Human  — unique ratio: {human_fp['unique_ratio']:.3f}, repeat rate: {human_fp['repeat_rate']:.3f}")
print(f"AI     — unique ratio: {ai_fp['unique_ratio']:.3f}, repeat rate: {ai_fp['repeat_rate']:.3f}")
print(f"Target — unique ratio: {target_fp['unique_ratio']:.3f}, repeat rate: {target_fp['repeat_rate']:.3f}")
print(f"\nAI Likelihood Score: {score:.1%}")
print(f"Verdict: {'LIKELY AI-GENERATED' if score > 0.6 else 'LIKELY HUMAN-WRITTEN' if score < 0.4 else 'UNCERTAIN'}")

# Top repeated trigrams in target:
print("\nTop repeated n-grams:")
for ngram, count in target_fp['top_ngrams']:
    print(f"  '{ngram}' — {count}×")
